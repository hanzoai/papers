\section{Multi-Chain Support}

\label{sec:multichain}
% ============================================================================

\subsection{EVM Chains (Lux, Ethereum)}

\begin{itemize}[leftmargin=1.4em]
    \item Address derivation: \texttt{keccak256(pubkey)[12:]} from MPC
    public key.
    \item Token support: native ETH/LUX + ERC-20 (USDC, USDT).
    \item Confirmation policy: 1 block on Lux (final), 12 blocks on
    Ethereum ($\sim$3 minutes).
    \item Gas estimation: real-time, paid from merchant's gas wallet
    (auto-funded from settlement).
\end{itemize}

\subsection{Bitcoin}

\begin{itemize}[leftmargin=1.4em]
    \item Address derivation: \texttt{bech32} (native SegWit P2WPKH) from
    compressed MPC public key.
    \item Confirmation policy: 1 confirmation for amounts $<$\$1,000, 3
    confirmations for $<$\$10,000, 6 confirmations for $\geq$\$10,000.
    \item Fee estimation: mempool-based fee rate selection targeting next-block
    inclusion.
    \item UTXO management: automated consolidation during low-fee periods.
\end{itemize}

\subsection{Solana}

\begin{itemize}[leftmargin=1.4em]
    \item Separate ed25519 MPC ceremony (CGGMP21 adapted for ed25519).
    \item SPL token support: USDC, USDT on Solana.
    \item Confirmation: single slot ($\sim$400ms).
    \item Transaction priority: Jito tips for guaranteed inclusion during
    congestion.
\end{itemize}
